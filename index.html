Objetivo: Desenvolver algoritmos para otimização de rota de um caminhão utilizando coordenadas cartesianas, aplicando a abordagem do Vizinho Mais Próximo e controlando o consumo de diesel.

# ===================================================================== # 11ª ATIVIDADE: OTIMIZAÇÃO DE ROTA E CONSUMO DE COMBUSTÍVEL # ===================================================================== import math def calcular_distancia(p1, p2): return math.sqrt((p2[0] - p1[0])**2 + (p2[1] - p1[1])**2) def calcular_rota_original(pontos): distancia_total = 0 ponto_atual = (0, 0) for p in pontos: distancia_total += calcular_distancia(ponto_atual, p) ponto_atual = p distancia_total += calcular_distancia(ponto_atual, (0, 0)) # Volta à origem return distancia_total def otimizar_rota_vizinho(pontos): nao_visitados = pontos.copy() rota_otimizada = [] ponto_atual = (0, 0) distancia_total = 0 while nao_visitados: proximo_ponto = min(nao_visitados, key=lambda p: calcular_distancia(ponto_atual, p)) distancia_total += calcular_distancia(ponto_atual, proximo_ponto) rota_otimizada.append(proximo_ponto) nao_visitados.remove(proximo_ponto) ponto_atual = proximo_ponto distancia_total += calcular_distancia(ponto_atual, (0, 0)) return rota_otimizada, distancia_total # Definição das coordenadas das entregas em km pontos_entrega = [(2, 3), (10, 5), (1, 8), (6, 2)] consumo_por_km = 0.35 capacidade_tanque = 80.0 dist_original = calcular_rota_original(pontos_entrega) rota_otim, dist_otimizada = otimizar_rota_vizinho(pontos_entrega) combustivel_original = dist_original * consumo_por_km combustivel_otim = dist_otimizada * consumo_por_km print("--- RESULTADO COMPARATIVO DE ROTAS ---") print(f"Rota Original: Sequência Padrão -> Distância: {dist_original:.2f} km | Consumo: {combustivel_original:.2f} L") print(f"Rota Otimizada: {rota_otim} -> Distância: {dist_otimizada:.2f} km | Consumo: {combustivel_otim:.2f} L") print(f"Economia de Combustível gerada: {combustivel_original - combustivel_otim:.2f} L") if combustivel_otim > capacidade_tanque: print("Alerta: O caminhão precisará reabastecer durante o percurso otimizado!") else: print("Sucesso: O tanque de 80L é suficiente para completar a rota otimizada.")

# ===================================================================== # 21ª e 22ª ATIVIDADE: MATRIZES DINÂMICAS E PERSISTÊNCIA DE ARQUIVOS # ===================================================================== def criar_malha_termica_dinamica(dimensao): """Gera uma matriz quadrada com dimensões dinâmicas informadas pelo operador""" print(f"\n--- Coleta de Dados para Malha Térmica de Tamanho {dimensao}x{dimensao} ---") malha = [] for i in range(dimensao): linha = [] for j in range(dimensao): temp = float(input(f"Digite a temperatura do ponto [{i}][{j}] (°C): ")) linha.append(temp) malha.append(linha) return malha def analisar_e_salvar_relatorio(matria_dados, nome_arquivo="relatorio_auditoria_termica.txt"): """Varre a matriz dinamicamente, processa estatísticas e escreve em arquivo externo""" total_pontos = len(matria_dados) * len(matria_dados[0]) soma_temperaturas = 0.0 maior_temperatura = matria_dados[0][0] for linha in matria_dados: for valor in linha: soma_temperaturas += valor if valor > maior_temperatura: maior_temperatura = valor media_geral = soma_temperaturas / total_pontos # Bloco para gravação segura e persistente de dados usando gerenciador de contexto (with) try: with open(nome_arquivo, "w", encoding="utf-8") as arquivo: arquivo.write("====================================================\n") arquivo.write(" RELATÓRIO TÉCNICO DE AUDITORIA TÉRMICA\n") arquivo.write("====================================================\n\n") arquivo.write(f"Dimensões da Malha Analisada: {len(matria_dados)}x{len(matria_dados)} pontos.\n") arquivo.write(f"Total de sensores mapeados em campo: {total_pontos}\n") arquivo.write(f"Temperatura média computada: {media_geral:.2f} °C\n") arquivo.write(f"Ponto térmico de maior pico medido: {maior_temperatura:.2f} °C\n\n") arquivo.write("--- MAPA COMPLETO DAS LEITURAS DA MALHA ---\n") for linha in matria_dados: linha_formatada = " | ".join([f"{val:.1f}°C" for val in linha]) arquivo.write(f"[ {linha_formatada} ]\n") arquivo.write("\n====================================================\n") print(f"\nSucesso: O relatório estruturado foi persistido e salvo em '{nome_arquivo}' com sucesso!") except IOError: print("\nErro de sistema: Não foi possível abrir ou gravar dados no arquivo externo especificado.") # Bloco Principal Interativo try: tamanho = int(input("Informe o tamanho desejado para a malha de sensores (ex: 3 para uma matriz 3x3): ")) if tamanho <= 0: print("Dimensão inválida. O tamanho da malha deve ser maior que zero.") else: malha_sensores = criar_malha_termica_dinamica(tamanho) analisar_e_salvar_relatorio(malha_sensores) except ValueError: print("Erro: A dimensão informada deve ser um número inteiro válido.")
