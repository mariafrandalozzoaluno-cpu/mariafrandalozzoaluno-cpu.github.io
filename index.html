# Exemplo de cálculo de tarifas baseado em faixas de consumo KWh
consumo = float(input("Consumo (KWh): "))
if consumo <= 100:
    tarifa = consumo * 0.50
else:
    tarifa = (100 * 0.50) + ((consumo - 100) * 0.85)
print(f"Tarifa calculada: R$ {tarifa:.2f}")

medicoes = [115.6, 220.1, 218.4, 114.2, 221.0, 118.9, 219.5]
soma_tensoes = sum(medicoes)

Exercício Atividades 10 a 15: Estruturas de Repetição Loops e Funções

Série de laboratórios focados na automação de processos repetitivos via estruturas de laço (`while` e `for`) e encapsulamento reutilizável de código (funções modulares).

Lógica de Loops Aplicada a Monitoramentos Contínuos:

# Simulação de Loop de Leitura Ininterrupta com parada de segurança
amostras = 0
limite_amostras = 5
while amostras < limite_amostras:
    print(f"Carregando dados da leitura elétrica sequencial nº {amostras + 1}")
    amostras += 1

# Exemplo de modularização por função para cálculo de Potência Ativa
def calcular_potencia_ativa(tensao, corrente, cosseno_phi):
    return tensao * corrente * cosseno_phi

p_final = calcular_potencia_ativa(220, 15, 0.92)
print(f"Potência Ativa Modular Calculada: {p_final:.2f} W")

# Mapeamento e Varredura de uma Matriz 2x2 para Sistemas Elétricos
matriz_impedancias = [
    [10, 5],
    [5, 12]
]
for i in range(len(matriz_impedancias)):
    for j in range(len(matriz_impedancias[i])):
        print(f"Impedância na posição [{i}][{j}]: {matriz_impedancias[i][j]} Ohms")
