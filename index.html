# Exercício 1: Tensão Segura
tensao = float(input("Digite a tensão medida (V): "))
if tensao > 220:
    print("Perigo: alta tensão!")
else:
    print("Tensão segura.")

# Exercício 2: Corrente em Resistor

    print("Sobrecarga!")
